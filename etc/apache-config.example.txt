# Apache config snippet for Heikki's beer tracker
# Edit the paths as necessary

# Copy or symlink to /etc/apache2/conf-available|enabled or conf.d
# Make also a symlink in /var/www/html/beertracker to the right place
# Create /etc/lsd/login.secret: sudo mkdir -p /etc/lsd && openssl rand -hex 32 | sudo tee /etc/lsd/login.secret && sudo chmod 640 /etc/lsd/login.secret && sudo chown root:www-data /etc/lsd/login.secret
# Create a .htpasswd (still used by login.pm for password validation)
# Create a directory called beerdata, Make it owned and writable by www-data.


<DirectoryMatch "^/var/www/html/beertracker(-dev)?/">
   AllowOverride All
   AcceptPathInfo On
   CGIPassAuth On
   Options +FollowSymLinks +ExecCGI -Indexes

   AddHandler cgi-script .cgi
   AddHandler fcgid-script .fcgi
   DirectoryIndex index.fcgi code/index.cgi

   # Authentication is handled by login.pm (cookie + HTTP Basic via CGIPassAuth).
   # No Apache-level auth required; access to the app is open at the Apache level.
   Require all granted
</DirectoryMatch>

# Tell the browser check image age every time - our graphs tend to be alive
# Needs a2enmod headers
<FilesMatch ".png$">
  Header set Cache-Control "no-cache"
</FilesMatch>

# Hide most of the data directory, but not the graphs we build there
<DirectoryMatch "^/var/www/html/beertracker(-dev)?/beerdata">
    Require all denied
    <FilesMatch "\.png$">
        Require all granted
        Satisfy any
    </FilesMatch>
</DirectoryMatch>

# Allow photos from the photo dir
<DirectoryMatch "^/var/www/html/beertracker(-dev)?/beerdata/.*\.photo">
    Require all granted
</DirectoryMatch>

# Allow static files
<DirectoryMatch "^/var/www/html/beertracker(-dev)?/static">
  Require all granted
  Satisfy any
</DirectoryMatch>

# Hide the .git directory
<DirectoryMatch "^/var/www/html/beertracker(-dev)?/.git">
    Require all denied
</DirectoryMatch>

# Hide the scripts directory
<DirectoryMatch "^/var/www/html/beertracker(-dev)?/scripts">
    Require all denied
</DirectoryMatch>
